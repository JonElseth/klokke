function oppdaterLiveStatusBoks(hoursStr, minutesStr) { const statusEl = document.getElementById('live-status-bar'); if (!statusEl) return; try { const nå = new Date(); const gjeldendeTime = nå.getHours(); const dagensNøkkel = nå.toLocaleDateString('sv-SE', {timeZone: 'Europe/Oslo'}); const dagensKalenderHendelser = globalEventsCache[dagensNøkkel] || []; let nesteAvtale = null; const sorterteHendelser = dagensKalenderHendelser .filter(ev => ev.computedStart) .sort((a, b) => new Date(a.computedStart).getTime() - new Date(b.computedStart).getTime()); for (let ev of sorterteHendelser) { const startTidspunkt = new Date(ev.computedStart); if (startTidspunkt > nå) { nesteAvtale = ev; break; } } let boksHTML = `Nå er klokken ${hoursStr}:${minutesStr}. `; if (gjeldendeTime >= 23 || gjeldendeTime < 7) { let erKritiskMorgenAvtale = false; if (nesteAvtale) { const tidTilNesteAvtale = new Date(nesteAvtale.computedStart).getTime() - nå.getTime(); const seksTimerIMillisekunder = 6 * 60 * 60 * 1000; if (tidTilNesteAvtale > 0 && tidTilNesteAvtale <= seksTimerIMillisekunder) { erKritiskMorgenAvtale = true; } } if (erKritiskMorgenAvtale) { const nesteTid = new Date(nesteAvtale.computedStart).toLocaleTimeString('no-NO', {hour: '2-digit', minute:'2-digit', timeZone: 'Europe/Oslo'}); let renSummary = nesteAvtale.summary ? nesteAvtale.summary.replace(/^(Mamma\s*&\s*Pappa:|Mamma\s+og\s+Pappa:|Mamma:|Pappa:)\s*/i, '') : ''; boksHTML += `Neste avtale er ${renSummary} klokken ${nesteTid}.`; } else { const iMorgen = new Date(); iMorgen.setDate(nå.getDate() + 1); const morgendagensNøkkel = iMorgen.toLocaleDateString('sv-SE', {timeZone: 'Europe/Oslo'}); const morgendagensHendelser = globalEventsCache[morgendagensNøkkel] || []; const førsteIAvtaleIMorgen = morgendagensHendelser .filter(ev => ev.computedStart) .sort((a, b) => new Date(a.computedStart).getTime() - new Date(b.computedStart).getTime())[0]; if (førsteIAvtaleIMorgen) { // NY LOGIKK: Sjekk først tekst for klokkeslett, så computedStart const tekstTid = hentKlokkeslettFraTekst(førsteIAvtaleIMorgen.summary); const morgenTime = tekstTid ? tekstTid.timer : new Date(førsteIAvtaleIMorgen.computedStart).getHours(); let periodeTekst = "i morgen"; if (morgenTime >= 7 && morgenTime < 9) periodeTekst = "i morgen tidlig"; else if (morgenTime >= 9 && morgenTime < 12) periodeTekst = "i morgen formiddag"; else if (morgenTime >= 12 && morgenTime < 18) periodeTekst = "i morgen ettermiddag"; else if (morgenTime >= 18 && morgenTime < 23) periodeTekst = "i morgen kveld"; boksHTML += `Det er natt, og det er ingen flere avtaler før ${periodeTekst}.`; } else { boksHTML += `Det er natt, og det er ingen flere avtaler i morgen.`; } } } else { if (nesteAvtale) { const nesteTid = new Date(nesteAvtale.computedStart).toLocaleTimeString('no-NO', {hour: '2-digit', minute:'2-digit', timeZone: 'Europe/Oslo'}); let renSummary = nesteAvtale.summary ? nesteAvtale.summary.replace(/^(Mamma\s*&\s*Pappa:|Mamma\s+og\s+Pappa:|Mamma:|Pappa:)\s*/i, '') : ''; boksHTML += `Neste avtale er ${renSummary} klokken ${nesteTid}.`; } else { boksHTML += `Ingen flere planlagte avtaler i dag.`; } } statusEl.innerHTML = boksHTML; } catch (feil) { statusEl.innerHTML = `Nå er klokken ${hoursStr}:${minutesStr}.`; } }
